# BNF for [pending language name]

<program> ::= BEGIN <code> END

<code> ::= <statement> <code>
				 | <statement>

<statement> ::= <assignment_statement>
							| <loop_statement>
							| <if_statement>
							| <rectangle_statement>
							| <pixel_statement>
							| <music_statement>

<assignment_statement> ::= VARIABLE ASSIGNMENT expression PERIOD

<loop_satement> ::= LOOPHEAD compound_condition BEGIN code END

<if_statement> ::= IFHEAD compound_condition BEGIN code END

<rectangle_statement> ::= RECTANGLE color_literal COMMA expression COMMA expression COMMA expression COMMA expression PERIOD

<pixel_statement> ::= PIXEL color_literal COMMA expression COMMA expression PERIOD

<music_statement> ::= MUSIC note_literal PERIOD

expression ::= factor
						 | expression PLUS factor
						 | expression MINUS factor

factor ::= term
				 | factor TIMES term
				 | factor DIVIDES term

term ::= variable
			 | number

compound_condition ::= condition
										 | condition AND condition
										 | condition OR condition

condition ::= EQUALS
						| GT
						| LT
						| GORE
						| LORE

color_literal ::= BLACK
								| WHITE
								| RED
								| CYAN
								| PURPLE
								| GREEN
								| BLUE
								| YELLOW
								| ORANGE
								| BROWN
								| LTRED
								| DKRED
								| GREY
								| LTGREEN
								| LTBLUE
								| LTGREY

note_literal ::= note
